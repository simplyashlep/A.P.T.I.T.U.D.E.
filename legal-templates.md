<script>
(function() {
  'use strict';
  var searchInput = document.getElementById('template-search');
  var cards = document.querySelectorAll('#template-grid .workspace-card');

  function filterCards() {
    var query = searchInput ? searchInput.value.trim().toLowerCase() : '';
    cards.forEach(function(card) {
      var text = card.textContent.toLowerCase();
      var category = (card.dataset.category || '').toLowerCase();
      var matches = !query || text.indexOf(query) !== -1 || category.indexOf(query) !== -1;
      card.style.display = matches ? '' : 'none';
    });
  }

  if (searchInput) searchInput.addEventListener('input', filterCards);
})();
</script>
